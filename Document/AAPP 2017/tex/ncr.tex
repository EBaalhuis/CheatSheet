// Calculate n choose k, also knows as n nCr k, or n above k
// Works for n <= 10 000.
static BigInteger ncr(int n, int k) {
	k = k > n / 2 ? n - k : k;
	BigInteger a = BigInteger.ONE;
	for (int i = 1; i < k + 1; i++) {
		a = a.multiply(BigInteger.valueOf
		(n - i + 1)).divide(BigInteger.valueOf(i));
	}
	return a;
}
